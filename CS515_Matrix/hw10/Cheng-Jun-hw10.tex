\mu  &= \frac{g_0^Tg}{g_0^T\mA g_0} \\
x_1 & = x_0 - \mu g_0 \\
g_1 & = \mA x_1-b = \mA ( x_0 - \mu g_0) - b  \\
& = \mA [ x_0-\frac{g_0^Tg}{g_0^T\mA g_0} g_0] -b \\
g_0^Tg_1 &= g_0^T[\mA x_0-b - \mA \frac{g_0^Tg}{g_0^T\mA g_0} g_0] \\
 & = g_0^T(\mA x_0 -b) - g_0^Tg_0  \\

\end{align*}
